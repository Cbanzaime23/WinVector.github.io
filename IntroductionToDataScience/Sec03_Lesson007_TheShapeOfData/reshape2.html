# load some libraries
library('ggplot2')
library('reshape2')
d <- data.frame(x=seq(0,10,by=0.05))
d$sinx <- sin(d$x)
d$sin1.5x <- sin(1.5*d$x)
d$cosx <- cos(d$x)
# plot it the hard way
ggplot(data=d) + 
  geom_line(aes(x=x,y=sinx)) +
  geom_line(aes(x=x,y=sin1.5x)) +
  geom_line(aes(x=x,y=cosx))
